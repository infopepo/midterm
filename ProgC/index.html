#include <stdlib.h>

int ggT(int, int);

    int a, b;
    scanf("%d%d", &a, &b);
    printf("ggT von a und b ist: %d", ggT(a, b));
    return 0;
}
int ggT(int a, int b) {
    int g;
    if (a == 0) return abs(b);
    if (b == 0) return abs(a);
    do {
        g = a / b;
        a = b;
        b = g;
    } while (b != 0);
    return abs(a);
}
